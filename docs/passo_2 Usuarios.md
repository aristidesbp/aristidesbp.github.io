# FUNCTION TRIGGER (codigo)
```
-- 1. Cria√ß√£o da fun√ß√£o que ser√° disparada
create or replace function public.handle_new_user()
returns trigger
language plpgsql
security definer set search_path = public
as $$
begin
  insert into public.usuarios (user_auth_users_id, email, nome)
  values (
    new.id, 
    new.email, 
    new.raw_user_meta_data->>'full_name' -- Captura o nome se enviado via metadata
  );
  return new;
end;
$$;

-- 2. Cria√ß√£o do Trigger propriamente dito
create trigger on_auth_user_created
  after insert on auth.users
  for each row execute procedure public.handle_new_user();
```

    // Fun√ß√£o para checar se a conex√£o j√° foi estabelecida (polling r√°pido)
    const checkSupabase = setInterval(async () => {
        if (window._supabase) {

            if (error || !session) {
                console.warn("Sess√£o inv√°lida, redirecionando...");

            // Opcional: Escuta mudan√ßas na autentica√ß√£o (ex: deslogar em outra aba)
            window._supabase.auth.onAuthStateChange((event) => {

        }
    }, 100);

    // Timeout de seguran√ßa: se ap√≥s 5s o supabase n√£o carregar, algo deu errado
    setTimeout(() => clearInterval(checkSupabase), 5000);
})();
