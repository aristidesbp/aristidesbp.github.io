CREATE TABLE public.entidades (
  id uuid NOT NULL DEFAULT gen_random_uuid(),
  created_at timestamp with time zone NOT NULL DEFAULT timezone('utc'::text, now()),
  user_id uuid DEFAULT auth.uid(),
  tipo text CHECK (tipo = ANY (ARRAY['Paciente'::text, 'Responsável'::text, 'Colaborador'::text])),
  nome_completo text NOT NULL,
  cpf text,
  data_nascimento date,
  email text,
  telefone text,
  cep text,
  logradouro text,
  numero text,
  bairro text,
  cidade text,
  uf text,
  permissao_ler boolean DEFAULT false,
  permissao_editar boolean DEFAULT false,
  permissao_cadastrar boolean DEFAULT false,
  permissao_deletar boolean DEFAULT false,
  urls_permitidas text,
  foto_url text,
  CONSTRAINT entidades_pkey PRIMARY KEY (id)
);
